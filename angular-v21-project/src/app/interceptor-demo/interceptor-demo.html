HTTP Interceptors

Two Interceptors Active

1. authTokenInterceptor

Clones every request and adds Authorization: Bearer <token>

export const authTokenInterceptor: HttpInterceptorFn = (req, next) => {
  const clonedReq = req.clone({
    setHeaders: { Authorization: `Bearer ${token}` }
  });
  return next(clonedReq);  // forward cloned request
};

2. loggingInterceptor

Times each request and logs method, URL, status, duration to RequestLogService

export const loggingInterceptor: HttpInterceptorFn = (req, next) => {
  const startTime = performance.now();
  return next(req).pipe(
    tap({ next: event => logService.logResponse(...) })
  );
};

Registered in app.config.ts

provideHttpClient(
  withFetch(),
  withInterceptors([
    authTokenInterceptor,
    loggingInterceptor,
  ])
)
